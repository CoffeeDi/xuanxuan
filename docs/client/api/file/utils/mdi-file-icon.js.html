/**
 * 根据文件名称获取对应的图标
 *
 * @function
 * @param {string} fileName 文件名
 * @return {string}
 */
export default fileName => {
    if (fileName.includes('.')) {
        const ext = fileName.substr(fileName.lastIndexOf('.') + 1);
        const icon = fileIcons[ext.toLowerCase()];
        if (icon) {
            return icon;
        }
    }
    return 'file-outline';
};
